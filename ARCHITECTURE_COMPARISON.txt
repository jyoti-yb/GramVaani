  _id: ObjectId (auto),
  email: string,

  language: string,
  location: string,
  created_at: datetime
}

user_queries collection:
{
  _id: ObjectId (auto),
  user_email: string,
  query: string,
  response: string,
  query_type: string,
  timestamp: datetime
}


┌─────────────────────────────────────────────────────────────────────────┐
│                         DynamoDB Tables                                  │
└─────────────────────────────────────────────────────────────────────────┘

gramvaani_users table:
{
  email: string (partition key),
  password: string (hashed),
  language: string,
  location: string,
  created_at: string (ISO format)
}

gramvaani_user_queries table:
{
  query_id: string (partition key, UUID),
  user_email: string (GSI),
  query: string,
  response: string,
  query_type: string,
  timestamp: string (ISO format)
}

╚══════════════════════════════════════════════════════════════════════════╝

For your use case (Gram Vaani - Rural AI Assistant):

✅ Use DynamoDB because:
   • Better performance for key-based lookups (user auth)
   • Auto-scaling for variable rural usage patterns
   • Lower cost at scale
   • Native AWS integration (you're already using AWS services)
   • Better for read-heavy workloads (queries)

⚠️  Stick with MongoDB if:
   • You need complex queries and aggregations
   • You prefer document-based flexibility
   • Your team is more familiar with MongoDB

💡 My Recommendation: Switch to DynamoDB
   Your app is perfect for DynamoDB's strengths!
